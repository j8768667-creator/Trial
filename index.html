#!/usr/bin/env python3
"""
riffle_builder.py

Experimental template-based .riffle generator for Riffler / Riffler Flow.

What it does:
- Reads a known-good .riffle file exported by the app.
- Transposes note-like values in the four 512-value music blocks.
- Optionally changes the BPM field.
- Writes a UTF-16 LE .riffle file that preserves the original structure.

This is not an official Riffler format implementation.
Always keep the original template and test generated files in the app.
"""

from __future__ import annotations

import argparse
from pathlib import Path
from typing import List

BLOCK_STARTS = (1024, 1536, 2048, 2560)
BLOCK_SIZE = 512

# Values in these blocks that look like MIDI pitches.
# Standard guitar range is approximately E2 (40) to E6 (88).
MIN_MIDI = 40
MAX_MIDI = 88

# Known chord roots in semitones, using C = 0.
ROOTS = {
    "C": 0, "C#": 1, "DB": 1,
    "D": 2, "D#": 3, "EB": 3,
    "E": 4,
    "F": 5, "F#": 6, "GB": 6,
    "G": 7, "G#": 8, "AB": 8,
    "A": 9, "A#": 10, "BB": 10,
    "B": 11,
}

def read_riffle(path: Path) -> List[str]:
    text = path.read_text(encoding="utf-16")
    values = text.split(",")
    if len(values) < 2944:
        raise ValueError(
            f"{path.name} has only {len(values)} fields; expected about 2945."
        )
    return values

def write_riffle(path: Path, values: List[str]) -> None:
    path.write_text(",".join(values), encoding="utf-16")

def signed_interval(source_root: str, target_root: str) -> int:
    source = ROOTS[source_root.upper()]
    target = ROOTS[target_root.upper()]
    delta = (target - source) % 12
    # Prefer the smallest movement.
    return delta - 12 if delta > 6 else delta

def transpose_blocks(values: List[str], semitones: int) -> int:
    changed = 0
    for start in BLOCK_STARTS:
        end = min(start + BLOCK_SIZE, len(values))
        for i in range(start, end):
            raw = values[i]
            try:
                n = int(raw)
            except ValueError:
                continue

            # Preserve common sentinels, booleans, counters and empty slots.
            if n in (-1, 0, 1, 2, 3, 4, 8, 9, 16, 32):
                continue

            if MIN_MIDI <= n <= MAX_MIDI:
                shifted = n + semitones
                if MIN_MIDI <= shifted <= MAX_MIDI:
                    values[i] = str(shifted)
                    changed += 1
    return changed

def set_bpm(values: List[str], bpm: int) -> None:
    if not 30 <= bpm <= 300:
        raise ValueError("BPM must be between 30 and 300.")

    # In the supplied 100 BPM samples, field 0 is 200.
    # Riffler appears to store BPM doubled here.
    values[0] = str(bpm * 2)

def main() -> None:
    parser = argparse.ArgumentParser(
        description="Create a transposed .riffle file from a known-good template."
    )
    parser.add_argument("template", type=Path, help="Source .riffle template")
    parser.add_argument("output", type=Path, help="Output .riffle filename")

    group = parser.add_mutually_exclusive_group(required=True)
    group.add_argument(
        "--semitones", type=int,
        help="Transpose by this many semitones, e.g. 7 for C to G"
    )
    group.add_argument(
        "--roots", nargs=2, metavar=("SOURCE", "TARGET"),
        help="Transpose from one root to another, e.g. --roots C G"
    )

    parser.add_argument("--bpm", type=int, help="Set BPM, e.g. 100")
    args = parser.parse_args()

    values = read_riffle(args.template)

    if args.roots:
        semitones = signed_interval(args.roots[0], args.roots[1])
    else:
        semitones = args.semitones

    changed = transpose_blocks(values, semitones)

    if args.bpm is not None:
        set_bpm(values, args.bpm)

    write_riffle(args.output, values)
    print(f"Created: {args.output}")
    print(f"Transposed note-like fields: {changed}")
    print(f"Interval: {semitones:+d} semitones")
    if args.bpm:
        print(f"BPM: {args.bpm}")
    print("Test the file in Riffler/Riffler Flow and keep your original template.")

if __name__ == "__main__":
    main()
